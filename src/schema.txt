CREATE DATABASE UbidTours;
USE UbidTours;

CREATE TABLE packages (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    pkgURL VARCHAR(500),
    numberOfDays VARCHAR(100),
    tourDetail TEXT,
    ageRange VARCHAR(100),
    isChildFriendly BOOLEAN,
    isHandicapAccessible BOOLEAN,
    insurancerequired BOOLEAN,
    promotionPercentage VARCHAR(50),
    seatsPerTour VARCHAR(50),
    groupSize VARCHAR(50),
    basePrice VARCHAR(100),
    singleSupplementPrc VARCHAR(100)
);

CREATE TABLE packageMood (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    mood VARCHAR(255),
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE subMood (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    sub_mood VARCHAR(255),
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE destinations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    destination VARCHAR(255),
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE pkgDays (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    day VARCHAR(50),
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE pkgMonths (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    month VARCHAR(50),
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE pkgYears (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    year VARCHAR(50),
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE pkgType (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    type VARCHAR(255),
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE tourPlan (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    dayNumber VARCHAR(50),
    detail TEXT,
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE numberTravelerPrice (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    rangeTitle VARCHAR(255),
    price INT,
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE mealSummary (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    mealType VARCHAR(255),
    detail TEXT,
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE transportation (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    mode VARCHAR(255),
    detail TEXT,
    FOREIGN KEY (package_id) REFERENCES packages(id)
);

CREATE TABLE transportationUpgrade (
    id INT AUTO_INCREMENT PRIMARY KEY,
    package_id INT,
    upgradeType VARCHAR(255),
    detail TEXT,
    FOREIGN KEY (package_id) REFERENCES packages(id)
);
